98  // No state change, do nothing
99  if (getJimulatorState() == newState) {
100  return;
101  }
102 
103  // Update the overall state of Jimulator State

105 
106  // Handles refreshing the views

113  case JimulatorState::UNLOADED:
114  refreshViews();
